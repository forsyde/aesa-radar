\newcommand{\drawcube}[1]{%
  \draw[very thick, fill=blue]%
  (#1) --++(0,-1,0) --++(-1,0,0) --++(0,1,0) -- cycle%
  (#1) --++(-1,0,0) --++(0,0,-1) --++(1,0,0) -- cycle%
  (#1) --++(0,-1,0) --++(0,0,-1) --++(0,1,0) -- cycle%
  ;%
}
